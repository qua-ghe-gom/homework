#bài tính điểm trung bình khi nhập điểm
#dạng nâng cao(hiểu thì làm k hiểu thì làm)
while True:
    try:
        toan=float(input("nhập điểm toán: "))
        ly=float(input("nhập điểm lý: "))
        hoa=float(input("nhập điểm hoá: "))
        if (toan>10 or ly >10 or hoa>10):
            print("vui lòng nhập điểm dưới 10")
        elif (toan<0 or ly <0 or hoa<0):
            print("điểm không được là số âm")
        else:
            print("điểm trung bình môn của toán lý hoá là: ",(toan*2+ly+hoa)/4)
            break
    except :
        print("vui lòng nhập số")
